```dataviewjs
// 1. 获取指定目录下的所有笔记（将 "你的文件夹路径" 替换为你想遍历的文件夹，若遍历全库则留空 ""）
const pages = dv.pages('"……"')
    .filter(p => p.keywords || p.knowledge_domains || p.functional_purposes || p.spiritual_directions); // 过滤掉没有这些字段的笔记

// 2. 生成列表
dv.list(pages.map(p => {
    // 格式化各个字段，如果字段不存在则显示“无”
    const keywords = p.keywords ? `**关键词**: ${p.keywords}` : '';
    const domains = p.knowledge_domains ? `**知识领域**: ${p.knowledge_domains}` : '';
    const purposes = p.functional_purposes ? `**功能目的**: ${p.functional_purposes}` : '';
    const directions = p.spiritual_directions ? `**精神导向**: ${p.spiritual_directions}` : '';
    
    // 将存在的字段过滤出来，并用换行和缩进连接，形成子列表
    const fieldsText = [keywords, domains, purposes, directions]
        .filter(text => text !== '')
        .join('\n    - '); // 四个空格加短横线，实现完美的子列表缩进
        
    // 返回：文章链接 + 换行子列表
    return `${p.file.link}\n    - ${fieldsText}`;
}));
```
