```dataviewjs
// 1. 辅助清洗函数：确保不管是数字、字符串还是空值，都转为干净的数字字符串
let clean = (v) => {
    if (v === undefined || v === null || v === "") return "";
    let s = String(v).replace(/[^0-9]/g, '');
    return s ? s : "";
};

// 2. 获取数据并直接在「链式流」中计算权重、完成排序
let sortedData = dv.pages()
    // 总筛选条件
    .where(p => p.book === "第一编 佛法總學" && p.sequence >= 1)
    // 核心逻辑：为每行数据动态组装时间、地点、刊名和排序权重
    .map(p => {
        let cy = clean(p.create_y), cm = clean(p.create_m), cd = clean(p.create_d);
        let py = clean(p.publish_y), pm = clean(p.publish_m), pd = clean(p.publish_d);
        
        let timeText = "";
        let isNormal = 1; // 1代表有时间，0代表无时间（用于沉底）
        let sortYear = "0000", sortMonth = "00", sortDay = "00";
        
        // 动态互斥字段备份
        let finalLocation = "—";
        let finalPublication = "—";
        
        // 时间格式化函数：根据有无月日动态决定拼接点
        let format = (y, m, d) => {
            if (!y) return "时间不详";
            let fY = y.padStart(4, '0');
            let fM = m ? m.padStart(2, '0') : "";
            let fD = d ? d.padStart(2, '0') : "";
            
            if (fM && fD) return `${fY}.${fM}.${fD}`; // yyyy.mm.dd
            if (fM) return `${fY}.${fM}`;             // yyyy.mm
            return fY;                                 // yyyy
        };

        // --- 核心级联与互斥逻辑 ---
        
        // 逻辑 3：创建仅有年，刊载有年月 -> 取刊载时间 (刊名填入，地点留空)
        if (cy && !cm && !cd && py && pm) {
            timeText = format(py, pm, pd);
            sortYear = py; sortMonth = pm; sortDay = pd || "00";
            finalPublication = p.publication || "—";
            finalLocation = "—"; 
        }
        // 逻辑 1：优先取创建时间 -> 取创建时间 (地点填入，刊名留空)
        else if (cy || cm || cd) {
            timeText = format(cy, cm, cd);
            sortYear = cy || "0000"; sortMonth = cm || "00"; sortDay = cd || "00";
            finalLocation = p.location || "—";
            finalPublication = "—"; 
        }
        // 逻辑 2：创建全空，取刊载 -> 取刊载时间 (刊名填入，地点留空)
        else if ((!cy && !cm && !cd) && (py || pm || pd)) {
            timeText = format(py, pm, pd);
            sortYear = py || "0000"; sortMonth = pm || "00"; sortDay = pd || "00";
            finalPublication = p.publication || "—";
            finalLocation = "—"; 
        }
        // 逻辑 4：全无时间
        else {
            timeText = "时间不详";
            isNormal = 0;
            sortYear = "9999"; 
            finalLocation = p.location || "—";
            finalPublication = p.publication || "—";
        }

        // 拼接成绝对纯数字权重，用于由旧到新排序
        let score = Number(`${sortYear.padStart(4,'0')}${sortMonth.padStart(2,'0')}${sortDay.padStart(2,'0')}`);

        // 将处理好的最终值绑定到页面对象
        p.computedTime = timeText;
        p.isNormal = isNormal;
        p.sortScore = score;
        p.computedLocation = finalLocation;
        p.computedPublication = finalPublication;
        return p;
    })
    // 强制第一层排序：先按是否有时间沉底 (1在前，0在后)
    .sort(p => p.isNormal, "desc")
    // 强制第二层排序：有时间的文件内部，按时间由旧到新排列 (asc)
    .sort(p => p.sortScore, "asc");

// 3. 渲染最终表格
dv.table(
    ["文章名", "时间", "地点", "刊名"],
    sortedData.map(p => [p.file.link, p.computedTime, p.computedLocation, p.computedPublication])
);
```

## 时间/地点/刊名列表（带优先级逻辑）

```dataviewjs
const currentFolder = dv.current().file.folder;
const pages = dv.pages('"' + currentFolder + '"')
  .where(function(p) {
    var name = p.file.name;
    return name && name.indexOf('仪表盘') === -1 && name.indexOf('編目錄') === -1;
  });

var withDate = [];
var withoutDate = [];

for (var i = 0; i < pages.length; i++) {
  var page = pages[i];
  var cy = page.create_y || '';
  var cm = page.create_m || '';
  var cd = page.create_d || '';
  var py = page.publish_y || '';
  var pm = page.publish_m || '';
  var pd = page.publish_d || '';
  var location = page.location || '';
  var publication = page.publication || '';
  var legacyDate = page.date || '';

  var hasCreate = !!(cy || cm || cd);
  var hasPublish = !!(py || pm || pd);

  var time = '';
  var loc = '';
  var pub = '';

  if (hasCreate) {
    if (cy && !cm && !cd && py && pm) {
      time = py + '年' + pm + '月' + (pd ? pd + '日' : '');
      pub = publication;
    } else {
      time = cy + '年' + (cm ? cm + '月' : '') + (cd ? cd + '日' : '');
      loc = location;
    }
  } else if (hasPublish) {
    time = py + '年' + (pm ? pm + '月' : '') + (pd ? pd + '日' : '');
    pub = publication;
  } else if (legacyDate) {
    var parts = legacyDate.split('-');
    time = parts[0] + '年' + (parts[1] ? parts[1] + '月' : '');
    loc = location;
    pub = publication;
  }

  if (time) {
    withDate.push({ time: time, loc: loc, pub: pub, link: page.file.link });
  } else {
    withoutDate.push({ loc: location, pub: publication, link: page.file.link });
  }
}

withDate.sort(function(a, b) {
  if (a.time < b.time) return 1;
  if (a.time > b.time) return -1;
  return 0;
});

var rows = [];
for (var j = 0; j < withDate.length; j++) {
  var d = withDate[j];
  rows.push([d.time, d.loc, d.pub, d.link]);
}
for (var k = 0; k < withoutDate.length; k++) {
  var u = withoutDate[k];
  rows.push(['无日期', u.loc, u.pub, u.link]);
}

dv.table(['时间', '地点', '刊名', '文章名'], rows);
```
